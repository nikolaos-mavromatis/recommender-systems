    Rule-based Recommender¶

    This is the most straightforward recommender system. Usually, it is based on rules defined by the business.

    In this case, I will be looking at the following rules:

    • Most popular items overall in terms of quantity sold,
    • Most popular items by country in terms of quantity sold, and
    • Buy again items for a specific user.

    Items are considered to be the unique stock codes.

    Retrieve popular items list¶

    def get_top_selling_items(data: pd.DataFrame, by: list[str] | None = None, top_n: int = 10) -&gt; pd.DataFrame:
    """
    Returns the top selling items in the data.

    Parameters
    ----------
    data : pd.DataFrame
        The data to analyze.
    by : list[str] | None, optional
        Additional columns to slice upon, by default None
    top_n : int, optional
        The number of items to return, by default 10

    Returns
    -------
    pd.DataFrame
        The top selling items.
    """
    grain = ["StockCode", "Description"]
    if by:
        grain += by

    temp = data.groupby(grain).agg({"Quantity": "sum"}).sort_values(["Quantity"], ascending=False)

    if by:
        return temp.groupby(level=by).head(top_n).reset_index(drop=False)

    return temp.head(top_n)

    Filter = dict[str, Union[str, int, float]]


def plot_top_selling_by_filter(data, filters: Filter | None = None, top_n: int = 10) -&gt; None:
    plot_data = data.copy()
    if filters:
        q = " and ".join([f"{k} == '{v}'" if isinstance(v, str) else f"{k} == {v}" for k, v in filters.items()])
        plot_data = data.query(q)

    _ = sns.barplot(
        x="Quantity",
        y="Description",
        data=plot_data,
        hue="Description",
        legend=False,
        palette="viridis",
    )
    _ = plt.suptitle(f"Top {top_n} selling items")
    if filters:
        _ = plt.title(
            f"Filtered by {', '.join([f'{k}: {v}' for k, v in filters.items()])}",
            fontsize=10,
        )
    _ = plt.xlabel("Quantity", fontdict={"fontweight": "bold"})
    _ = plt.ylabel("")
    plt.show()

    Buy again items¶

    def get_buy_again_list_by_customer_id(data: pd.DataFrame, customer_id: int, top_n: int = 10) -&gt; list[str]:
    """
    Returns the top selling items for a given customer.

    Parameters
    ----------
    data : pd.DataFrame
        The data to query.
    customer_id : int
        The customer ID to filter upon.
    top_n : int, optional
        The number of items to return, by default 10

    Returns
    -------
    pd.DataFrame
        A list of the last items bought by the given customer.
    """
    customer_data = data.query(f"CustomerID == {customer_id}")
    return list(Counter(customer_data["Description"]))[:top_n]
